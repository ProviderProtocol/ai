  Changes since v0.0.22

  Breaking Changes

  - Enum constants replace magic strings - All discriminator string literals are now exported as const objects for type-safe usage

  Migration Guide

  Replace string literals with enum constants:

```ts
  // Before
  if (event.type === 'text_delta') { ... }
  if (error.code === 'RATE_LIMITED') { ... }
  if (block.type === 'text') { ... }
  if (message.type === 'user') { ... }
```

```ts
  // After
  import { StreamEventType, ErrorCode, ContentBlockType, MessageRole } from 'upp';

  if (event.type === StreamEventType.TextDelta) { ... }
  if (error.code === ErrorCode.RateLimited) { ... }
  if (block.type === ContentBlockType.Text) { ... }
  if (message.type === MessageRole.User) { ... }
```
